Instance internalize_P : computable P.
Proof.
  change P with
      (fun (R : SRS) (x0 : Definitions.string) (a0 : symbol) =>
         R ++
           map (card1 a0) (a0 :: x0 ++ sym R) ++ map (card2 a0) (a0 :: x0 ++ sym R)).
  extract.
Defined.
